        Django Template Filters
        Syntax:
            { { variable_name | filter_name } }
    
        length filter
        { { variable_name | length } }
        
length of {{list}} is {{list|length}}

    
        add filter
        { { variable_name | add:"2" } }
        
befor filter val={{val}} after filter val={{val|add:"2"}}

    
        addslashes filter
        It is used to add slashes before quotes.
        { { value | addslashes } }
        
befor filter value={{addslash}} after filter value={{addslash|addslashes}}

    
        capfirst filter
        It is used to capitalize the first character of the value.
        { { value | capfirst } }
        
befor filter value={{cap}} after filter value={{cap|capfirst}}

    
        center filter
        { { value | center:'15' } }
        
befor filter value={{cap}} after filter value={{cap|center:'15'}}

    
        cut filter
        { { value | cut:" " } }
        
befor filter value="{{c}}" after filter value="{{c|cut:" "}}"

    
        default filter
        { { value | default:"nothing" } }
        
befor filter value="{{d}}" after filter value="{{d|default:"nothing"}}"

    
        dictsort filter
        { { value | dictsort:"name" } }
        
befor filter value="{{di}}" after filter value="{{di | dictsort:"name"}}"
